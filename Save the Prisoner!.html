int main(int argc, char const *argv[]) {
  int t;
  scanf("%d", &t);
  for (int i_t; i_t < t; i_t++) {
    long long n, m, s;
    scanf("%lld %lld %lld", &n, &m, &s);

    long long res = ((m + s - 1) % n);
    if (res == 0) res = n;

    printf("%lld\n", res);
  }
  return 0;
}

        {
            int N = input.nextInt();
            int M = input.nextInt();
            int S = input.nextInt();

            int poisonedPrisoner = (S + M - 1) % N;
            if(poisonedPrisoner == 0){poisonedPrisoner=N;}

        }
    }
