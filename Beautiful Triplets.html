  using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int n,d,total=0;
    cin>>n>>d;
    int a[n];
    for(int i=0;i< n;i++) {
        cin>>a[i];
    }
    for(int i = 0; i < n; i++){
        for(int j = i+1; j < n; j++){
            if(a[j] - a[i] == d){
                for(int k = j+1; k < n; k++){
                    if(a[k] - a[j] == d){
                        total++;
                    }
                }
            }
        }
    }
    cout<< total;
    return 0;
}

  #!/bin/python3

  import sys

  def beautifulTriplets(d, arr):
      res = 0

      for el in arr:
          if el + d in arr and el + 2*d in arr:
              res += 1

      return res

  if __name__ == "__main__":
      n, d = input().strip().split(' ')
      n, d = [int(n), int(d)]
      arr = list(map(int, input().strip().split(' ')))
      result = beautifulTriplets(d, arr)
      print(result)
