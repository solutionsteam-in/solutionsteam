int main(){

    int q;

    cin >> q;
    for(int i = 0; i < q; i++){
        int a,b,c;
        cin >> a >> b >> c;
        abs(a-c) < abs(b-c) ? cout << "Cat A" : abs(b-c) < abs(a-c) ? cout << "Cat B" : cout << "Mouse C";
        cout << endl;
    }
    return 0;
}

  #!/bin/python3

  import sys

  def catAndMouse(x, y, z):
      cat_a = abs(x - z)
      cat_b = abs(y - z)

      if cat_a < cat_b:
          return "Cat A"
      elif cat_a > cat_b:
          return "Cat B"
      else:
          return "Mouse C"

  if __name__ == "__main__":
      q = int(input().strip())
