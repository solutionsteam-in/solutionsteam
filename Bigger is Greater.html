                  while(flag==0 && j!=i-1){
                      if(str[i-1]< str[j])
                      {
                          char t = str[j];
                          str[j] = str[i-1];
                          str[i-1] = t;
                          flag = 1;
                          break loop;

  #!/bin/python3

  import sys
  from itertools import permutations

  def biggerIsGreater(w):
      arr = list(w)
      # Find non-increasing suffix
      i = len(arr) - 1
      while i > 0 and arr[i - 1] >= arr[i]:
              i -= 1
      if i <= 0:
      	return 'no answer'

      # Find successor to pivot
      j = len(arr) - 1
      while arr[j] <= arr[i - 1]:
          j -= 1
      arr[i - 1], arr[j] = arr[j], arr[i - 1]

      # Reverse suffix
      arr[i : ] = arr[len(arr) - 1 : i - 1 : -1]

      return "".join(arr)
