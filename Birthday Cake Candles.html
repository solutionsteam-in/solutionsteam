  import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);

        int n = scan.nextInt();
                int maxHeight = 1;
                int countMax = 0;

        for(int i = 0; i < n; i++) {
            int tmp = scan.nextInt();

            if(tmp > maxHeight) {
                maxHeight = tmp;
                countMax = 1;
            }
            else if(tmp == maxHeight) {
                countMax++;
            }
        }
        scan.close();

        System.out.println(countMax);
    }
}
