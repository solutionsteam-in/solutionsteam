using namespace std;

int main()
{
    int s,t,a,b,m,n,ap,apcnt = 0,ora,orcnt = 0;

    cin >> s >> t >> a >> b >> m >> n;

    for(int i = 0;i < m;i++)
 {
        cin >> ap;
        if(a+ap >= s && a+ap <= t)apcnt++;
    }

    for(int i = 0;i < n;i++)
 {
        cin >> ora;
        if(b+ora >= s && b+ora <= t)orcnt++;
    }
    cout << apcnt << endl<< orcnt << endl;
    return 0;
}

  import java.util.Scanner;

public class GradingStudents {

    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
        int n = in.nextInt();
        int arr[] = new int[n];
        for (int i = 0; i < n; i++) {
            int grade = in.nextInt();
            if (grade < 38) {
                arr[i] = grade;
            } else {
                int f1 = grade + 1;
                int f2 = grade + 2;
                if (f1 % 5 == 0) {
                    arr[i] = f1;
                } else if (f2 % 5 == 0) {
                    arr[i] = f2;
                } else {
                    arr[i] = grade;
                }
            }
        }
        for (int i = 0; i < n; i++) {
            System.out.println(arr[i]);
        }
    }
}

  #!/bin/python3

  import sys

  def appleAndOrange(s, t, a, b, apple, orange):
      count_app = 0
      count_org = 0
      for el in apple:
          if s <= el+a <= t:
              count_app += 1

      for el in orange:
          if s <= el+b <= t:
              count_org += 1

      return [ count_app, count_org ]
