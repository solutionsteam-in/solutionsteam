  #include < bits/stdc++.h>
using namespace std;
int main() {
    int n,updown = 0,res = 0;
    char c;
    cin >> n;

    for(int i = 0; i++ < n;){
        cin >> c;
        if(c == 'U')updown++;
        else updown--;
        if(updown == 0 && c == 'U')res++;
    }
    cout << res << endl;
    return 0;
}

JAVA:

  import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int level = 0;
        int prevLevel = 0;
        int counter = 0;
        String path = sc.next();
        for (int i = 0; i < path.length(); i++) {
            prevLevel = level;
            if (path.charAt(i) == 'U') {
                level++;
            } else {
                level--;
            }

            if (level == -1 && prevLevel == 0) {
                counter++;
            }
        }

        System.out.println(counter);
    }
}

  #!/bin/python3

  import sys

  def countingValleys(n, s):
      res = 0
      in_valley = 0
      curr = 0

      for step in s:
          if step == 'U':
              curr += 1
          else:
              curr -= 1

          if curr < 0 and in_valley == 0:
              in_valley = 1
          if in_valley == 1 and curr == 0:
              in_valley = 0
              res += 1

      return res

  if __name__ == "__main__":
      n = int(input().strip())
      s = input().strip()
      result = countingValleys(n, s)
      print(result)
