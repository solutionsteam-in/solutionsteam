  public static int diagonalDifference(List< List< Integer>> arr) {
      int leftd=0;
      int rightd=0;
      int n=arr.size();
      for(int i=0;i< n;i++)
      {
          leftd+=arr.get(i).get(i);
          rightd+=arr.get(i).get(n-i-1);
      }
      int absd=Math.abs(leftd-rightd);
      return absd;

  #!/bin/python3

  import sys

  def diagonalDifference(a):
      # compute primary diagonal
      n = len(a)
      diag_pr = diag_sec = 0
      for i in range(n):
          diag_pr += a[i][i]
          diag_sec += a[n-i-1][i]

      return abs(diag_pr - diag_sec)
